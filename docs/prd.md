#### **1. `lib/types.ts` (Type Definitions)**

// Konfigurasi utama
export interface SiteConfig {
  site: {
    title: string;
    description?: string;
    favicon?: string;
  };
  theme?: {
    primary?: string;
    secondary?: string;
    background?: string;
    text?: string;
    mode?: 'light' | 'dark';
  };
  navigation?: Array<{
    label: string;
    path: string;
  }>;
  pages: Array<{
    path: string;           // '/' atau '/section'
    type: string;           // 'hero', 'grid', 'cards', 'status', 'markdown'
    content: any;           // Konten spesifik untuk setiap type
    metadata?: {
      title?: string;
      description?: string;
    };
  }>;
}

// Type untuk Hero section
export interface HeroContent {
  title: string;
  subtitle?: string;
  description?: string;

    text: string;
    href: string;
    variant?: 'primary' | 'secondary' | 'outline';
  };
  backgroundImage?: string;
}

export interface GridContent {
  title?: string;
  subtitle?: string;

#### **2. `lib/config.ts` (Config Loader)**
```typescript
import { promises as fs } from 'fs';
import { join } from 'path';
import yaml from 'js-yaml';
import { SiteConfig } from './types';

const DEFAULT_CONFIG: SiteConfig = {
  site: {
    title: 'Engineering Dashboard',
    description: 'Generated by AI-powered Backstage'
  },
  pages: [
    {
      path: '/',
      type: 'hero',
      content: {
        title: 'Welcome to Your Engineering Hub',
        subtitle: 'Your services, tools, and documentation in one place',
        cta: {
          text: 'Get Started',
          href: '#services'
        }
      }
    }
  ]
};

export async function loadSiteConfig(): Promise<SiteConfig> {
  try {
    // Cek config.yaml dulu
    const configPath = join(process.cwd(), 'config.yaml');
    const configContent = await fs.readFile(configPath, 'utf-8');
    return yaml.load(configContent) as SiteConfig;
  } catch (yamlError) {
    try {
      // Fallback ke config.json
      const jsonPath = join(process.cwd(), 'config.json');
      const jsonContent = await fs.readFile(jsonPath, 'utf-8');
      return JSON.parse(jsonContent) as SiteConfig;
    } catch (jsonError) {
      console.warn('No config file found, using default configuration');
      return DEFAULT_CONFIG;
    }
  }
}

// Untuk development: watch config changes
export function setupConfigWatcher() {
  if (process.env.NODE_ENV === 'development') {
    const chokidar = require('chokidar');
    const watcher = chokidar.watch(['config.yaml', 'config.json'], {
      cwd: process.cwd(),
      ignoreInitial: true
    });

    watcher.on('change', () => {
      console.log('Config changed, reloading...');
      // Di development, kita bisa trigger refresh via React context atau event
    });
  }
}
```

#### **4. `app/page.tsx` (Dynamic Page Renderer)**
```tsx
import { loadSiteConfig } from '@/lib/config';
import { HeroSection } from '@/components/dynamic/Hero';
import { GridSection } from '@/components/dynamic/GridSection';
import { CardSection } from '@/components/dynamic/CardSection';
import { StatusSection } from '@/components/dynamic/StatusSection';
import { MarkdownSection } from '@/components/dynamic/MarkdownSection';

// Component mapping berdasarkan type di config
const COMPONENT_MAP = {
  hero: HeroSection,
  grid: GridSection,
  cards: CardSection,
  status: StatusSection,
  markdown: MarkdownSection,
};

export default async function HomePage() {
  const config = await loadSiteConfig();
  
  // Cari halaman untuk root path '/'
  const homePage = config.pages.find(page => page.path === '/');
  
  if (!homePage) {
    return (
      <div className="min-h-[400px] flex items-center justify-center">
        <div className="text-center">
          <h2 className="text-2xl font-bold mb-2">No Content Found</h2>
          <p className="text-muted-foreground">
            Please add a page with path &quot;/&quot; in your config.yaml
          </p>
        </div>
      </div>
    );
  }

  const Component = COMPONENT_MAP[homePage.type] || (() => (
    <div className="p-8 bg-muted rounded-lg">
      <h3 className="text-lg font-medium mb-2">Unknown Section Type</h3>
      <p>Type: {homePage.type}</p>
    </div>
  ));
  
  return (
    <div className="w-full">
      <Component content={homePage.content} />
      
      {/* Render semua sections lainnya sebagai bagian dari single page */}
      {config.pages
        .filter(page => page.path !== '/')
        .map((section, index) => {
          const SectionComponent = COMPONENT_MAP[section.type] || (() => (
            <div key={index} className="mb-12">
              <h3 className="text-xl font-bold mb-2">Section {index + 1}</h3>
              <div className="p-4 bg-muted rounded">
                <pre className="text-sm text-muted-foreground">
                  {JSON.stringify(section.content, null, 2)}
                </pre>
              </div>
            </div>
          ));
          
          return (
            <div key={index} id={section.path.replace('/', '')} className="mb-12">
              <SectionComponent content={section.content} />
            </div>
          );
        })
      }
    </div>
  );
}
```

```bash

npm run start
```
- Di production, config di-load saat build time
- Hasilnya adalah static website yang sudah tergenerate
